// ================== КОНФИГУРАЦИЯ КНОПОК ==================
const int buttonPins[] = {4, 5, 6, 7, 8, 9, 15, 16, 17, 18};
const int numButtons = 10;

// Переменные для управления мышью
int mouseX = 0;
int mouseY = 0;
const int mouseStep = 10;  // Шаг перемещения мыши

bool buttonPressed[numButtons] = {false};
unsigned long lastDebounceTime[numButtons] = {0};
const unsigned long debounceDelay = 50;

// ================== ПРОТОТИПЫ ФУНКЦИЙ ==================
void executeMacro(int buttonIndex);
void executeKeyCommand(uint8_t key, int param1, int param2);

void executeCtrlC();

void mouseMoveToOrigin();  // Функция для перемещения мыши в начало координат

// ================== НАСТРОЙКА ==================
void setup() {
  Serial.begin(115200);
  
  for (int i = 0; i < numButtons; i++) {
    pinMode(buttonPins[i], INPUT_PULLUP);
  }
  
  Keyboard.begin();
  Mouse.begin();
  USB.begin();

  WiFi.softAP(ssid, password);
  
  IPAddress local_ip(10, 0, 0, 1);
  IPAddress gateway(10, 0, 0, 1);
  IPAddress subnet(255, 255, 255, 0);
  WiFi.softAPConfig(local_ip, gateway, subnet);

  if (!MDNS.begin("pavlik")) {
    Serial.println("Error setting up MDNS responder!");
  }

  server.on("/", handleRoot);
  server.on("/macro", handleMacro);
  server.on("/status", handleStatus);
  
  server.begin();
  Serial.println("HTTP server started");
  Serial.print("AP IP address: ");
  Serial.println(WiFi.softAPIP());
}

// ================== ГЛАВНЫЙ ЦИКЛ ==================
void loop() {
  server.handleClient();
  
  for (int i = 0; i < numButtons; i++) {
    int buttonState = digitalRead(buttonPins[i]);
    
    if (buttonState == LOW) {
      if (millis() - lastDebounceTime[i] > debounceDelay) {
        if (!buttonPressed[i]) {
          executeMacro(i);
          buttonPressed[i] = true;
          lastDebounceTime[i] = millis();
        }
      }
    } else {
      buttonPressed[i] = false;
    }
  }
  
  delay(5);
}

// ================== ФУНКЦИЯ ПЕРЕМЕЩЕНИЯ МЫШИ В НАЧАЛО КООРДИНАТ ==================
void mouseMoveToOrigin() {
  const int steps = 5;
  const int stepDelay = 50;
  
  for (int i = 0; i < steps; i++) {
    Mouse.move(-100, -100);
    delay(stepDelay);
  }
  
  // Сбрасываем внутренние координаты
  mouseX = 0;
  mouseY = 0;
}

void handleMacro() {
  if (server.hasArg("btn")) {
    String btnArg = server.arg("btn");
    
    // Обработка команд мыши из веб-интерфейса
    if (btnArg == "up") {
      Mouse.move(0, -mouseStep);
      server.send(200, "text/plain", "Mouse moved up");
      return;
    } else if (btnArg == "down") {
      Mouse.move(0, mouseStep);
      server.send(200, "text/plain", "Mouse moved down");
      return;
    } else if (btnArg == "left") {
      Mouse.move(-mouseStep, 0);
      server.send(200, "text/plain", "Mouse moved left");
      return;
    } else if (btnArg == "right") {
      Mouse.move(mouseStep, 0);
      server.send(200, "text/plain", "Mouse moved right");
      return;
    } else if (btnArg == "click") {
      Mouse.click();
      server.send(200, "text/plain", "Mouse clicked");
      return;
    }
    
    // Обработка обычных макросов
    int buttonIndex = btnArg.toInt();
    if (buttonIndex >= 0 && buttonIndex < numButtons) {
      executeMacro(buttonIndex);
      server.send(200, "text/plain", "Macro " + String(buttonIndex) + " executed");
    }
  }
}

  status += "\nMouse position: X=" + String(mouseX) + ", Y=" + String(mouseY);
  server.send(200, "text/plain", status);
}

// ================== ФУНКЦИИ ДЛЯ CTRL+C/V ==================
void executeCtrlC() {
  Keyboard.press(KEY_LEFT_CTRL);
  Keyboard.press('c');
  delay(50);
  Keyboard.release('c');
  Keyboard.release(KEY_LEFT_CTRL);
  delay(100);
}

void executeCtrlV() {
  Keyboard.press(KEY_LEFT_CTRL);
  Keyboard.press('v');
  delay(50);
  Keyboard.release('v');
  Keyboard.release(KEY_LEFT_CTRL);
  delay(100);
}

// ================== ВСПОМОГАТЕЛЬНЫЕ ФУНКЦИИ ==================
void executeMacro(int buttonIndex) {
  // Для кнопки 0 - отправка текстовой команды
  if (buttonIndex == 0) {
    Keyboard.println(text_commands[buttonIndex]);
    return;
  }
  
  // Для кнопки 4 - Ctrl+C
  if (buttonIndex == 4) {
    executeCtrlC();
    return;
  }
  
  // Для кнопки 5 - Ctrl+V
  if (buttonIndex == 5) {
    executeCtrlV();
    return;
  }
  
  // Остальные кнопки - обычные макросы
  for (int i = 0; i < MAX_COMMANDS_PER_BUTTON; i++) {
    KeyCommand cmd = macro_commands[buttonIndex][i];
    if (cmd.key == COMMAND_TERMINATOR) break;
    
    executeKeyCommand(cmd.key, cmd.param1, cmd.param2);
  }
}

void executeKeyCommand(uint8_t key, int param1, int param2) {
  // Обработка команды перемещения мыши в начало координат
  if (key == MOUSE_TO_ORIGIN) {
    mouseMoveToOrigin();
    if (param2 > 0) delay(param2);
    return;
  }
  
  // Обработка команды относительного перемещения мыши
  if (key == MOUSE_MOVE_RELATIVE) {
    Mouse.move(param1, param2);  // param1 = X, param2 = Y
    mouseX += param1;
    mouseY += param2;
    return;
  }
  
  // Обработка команды абсолютного перемещения мыши
  if (key == MOUSE_MOVE_ABSOLUTE) {
    Mouse.move(param1 - mouseX, param2 - mouseY);
    mouseX = param1;
    mouseY = param2;
    return;
  }
  
  // Обработка клика мыши
  if (key == MOUSE_CLICK) {
    if (param1 == 1) {
      Mouse.click(MOUSE_LEFT);
    } else if (param1 == 2) {
      Mouse.click(MOUSE_RIGHT);
    } else if (param1 == 3) {
      Mouse.click(MOUSE_MIDDLE);
    }
    if (param2 > 0) delay(param2);
    return;
  }
  
  // Обработка задержки
  if (key == 0 && param1 == 0) {
    delay(param2);
    return;
  }
  
  // Стандартная обработка клавиш
  if (param1 > 0) {
    for (int j = 0; j < param1; j++) {
      Keyboard.press(key);
      Keyboard.release(key);
      if (param2 > 0 && j < param1 - 1) delay(param2);
    }
    if (param2 > 0) delay(param2);
  }
}
